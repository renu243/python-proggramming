#remove even numbers program
l=[]
n = int(input("Enter the Total List Items = "))
for i in range(1, n+1):
    l.append(i)
print("List Items = ", l)

l= [i for i in l if i % 2 != 0]    
print("List Items after removing even Items = ", l)
OUTPUT:
    Enter the Total List Items = 9
List Items =  [1, 2, 3, 4, 5, 6, 7, 8, 9]
List Items after removing even Items =  [1, 3, 5, 7, 9]
